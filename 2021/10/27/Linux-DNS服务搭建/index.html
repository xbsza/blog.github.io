01_DNS服务

发表于 2021-10-27 更新于 2021-10-29 分类于 应用服务 阅读次数： Valine：
本文字数： 7.6k 阅读时长 ≈ 7 分钟

  域名系统(Domain Name System,DNS)是Internet上解决网上机器命名的一种系统。就像拜访朋友要先知道别人家怎么走一样，Internet上当一台主机要访问另外一台主机时，必须首先获知其地址，TCP/IP中的IP地址是由四段以“.”分开的数字组成(此处以IPv4的地址为例，IPv6的地址同理)，记起来总是不如名字那么方便。所以，就采用了域名系统来管理名字和IP的对应关系。

环境安装

实验环境：Centos 7 64位
镜像：ISO镜像下载地址
实验需求：通过了解DNS原理以及配置方法，安装DNS服务实现正反向解析得到域名或者对应的IP地址。

[root@localhost ~]# cp /etc/named.conf /etc/named.conf.bak // 备份
[root@localhost ~]# vim /etc/named.conf

options {
listen-on port 53 { any; }; // 监听地址为所有地址 即 any
listen-on-v6 port 53 { ::1; };
directory “/var/named”;
dump-file “/var/named/data/cache_dump.db”;
statistics-file “/var/named/data/named_stats.txt”;
memstatistics-file “/var/named/data/named_mem_stats.txt”;
recursing-file “/var/named/data/named.recursing”;
secroots-file “/var/named/data/named.secroots”;
allow-query { any; }; // 允许网段为所有

[root@localhost ~]#

[root@localhost ~]# vim /etc/named.rfc1912.zones

zone “sk.com” IN { // 区域名称为sk..com
type master;
file “sk.com.zone”; // 正向解析的文件名称
allow-update { none; };
};

zone “67.168.192.in-addr.arpa” IN {
type master;
file “67.168.192.in-addr”; // 反向解析的文件名称
allow-update { none; };
};

[root@localhost ~]#

[root@localhost ~]# cp /var/named/named.localhost /var/named/sk.com.zone
[root@localhost ~]# cp /var/named/named.loopback /var/named/67.168.192.in-addr

[root@localhost ~]# chgrp named /var/named/sk.com.zone
[root@localhost ~]# chgrp named /var/named/67.168.192.in-addr
[root@localhost ~]# chmod 777 /var/named/sk.com.zone
[root@localhost ~]# chmod 777 /var/named/67.168.192.in-addr

[root@localhost ~]# vim /var/named/sk.com.zone

$TTL 1D
@ IN SOA dns.sk.com. root.sk.com. (
0 ; serial
1D ; refresh
1H ; retry
1W ; expire
3H ) ; minimum
@ IN NS dns.sk.com.
www IN A 192.168.67.130 // 主机记录为www
dns IN A 192.168.67.130
[root@localhost ~]#

[root@localhost ~]# vim /var/named/67.168.192.in-addr

$TTL 1D
@ IN SOA dns.sk.com. root.sk.com. (
0 ; serial
1D ; refresh
1H ; retry
1W ; expire
3H ) ; minimum
@ IN NS dns.sk.com.
130 IN PTR www.sk.com.
130 IN PTR dns.sk.com.
[root@localhost ~]#
